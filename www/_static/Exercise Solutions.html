"""
Problem 1.2 - Fibonacci Sequence
Author: Thomas Robitaille
"""

# Start of with the sequence consisting of [0, 1]
seq = [0, 1]

# Loop while the next element is below 100000
while seq[-1] + seq[-2] < 100000:

    # Find the next element by summing the two previous ones
    seq.append(seq[-1] + seq[-2])

# Print out the sequence:
print "Fibonacci Sequence:", seq

# Find values in the sequence that are squares
squares = []
for s in seq:
    for i in range(s + 1):
        if s == i * i:
            squares.append(s)

# Print out the squares
print "Squares:", squares

def factorial(n):
    prod = 1
    for i in range(2, n+1):
        prod *= i  # this is the same as prod = prod * i
    return prod

def factorial(n):
    if n == 0 or n == 1:
        return 1
    else:
        return n * factorial(n-1)

import math
print math.factorial(15)
print factorial(15)

class Particle(object):
    
    def __init__(self, mass, x, y, z, vx, vy, vz):
        self.mass = mass
        self.x = x
        self.y = y
        self.z = z
        self.vx = vx
        self.vy = vy
        self.vz = vz
        
    def kinetic_energy(self):
        return 0.5 * self.mass * (self.vx ** 2 + self.vy ** 2 + self.vz ** 2)
    
    def separation(self, other):
        return np.sqrt((self.x - other.x) ** 2 + (self.y - other.y) ** 2 + (self.z - other.z) ** 2)
    
    def move(self, dt):
        self.x += dt * self.vx
        self.y += dt * self.vy
        self.z += dt * self.vz
